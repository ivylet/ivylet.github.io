                    package com.taytay.demo;

import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;

public class MyProxy implements InvocationHandler {
    private UserDao userDao;
    public Object createProxy(UserDao userDao){
        this.userDao = userDao;
        ClassLoader classLoader = MyProxy.class.getClassLoader();
        Class[] classes = userDao.getClass().getInterfaces();
        return Proxy.newProxyInstance(classLoader,classes,this);
    }
    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        MyAspect myAspect = new MyAspect();
        myAspect.name();
        Object obj = method.invoke(userDao,args);
        myAspect.worry();
        return obj;
    }
}

                    package com.taytay.demo;

public class Test {
    public static void main(String[] args) {
        MyProxy myProxy = new MyProxy();
        UserDao userDao = new UserDaoImpl();
        UserDao userDao1 = (UserDao) myProxy.createProxy(userDao);
        userDao1.say();
        userDao1.addmit();
    }
}

                    package com.taytay.demo2;

import org.springframework.cglib.proxy.Enhancer;
import org.springframework.cglib.proxy.MethodInterceptor;
import org.springframework.cglib.proxy.MethodProxy;;
import java.lang.reflect.Method;

public class CglibProxy implements MethodInterceptor {

    public Object creatProxy(Object target){
        Enhancer enhancer = new Enhancer();
        enhancer.setSuperclass(target.getClass());
        enhancer.setCallback(this);
        return enhancer.create();
    }
    public Object intercept(Object proxy, Method method, Object[] args, MethodProxy methodProxy) throws Throwable{
        MyAspect myAspect = new MyAspect();
        myAspect.name();
        Object obj = methodProxy.invokeSuper(proxy,args);
        myAspect.worry();
        return obj;
    }
}

                    package com.taytay.demo2;

public class Test {
    public static void main(String[] args) {
        CglibProxy cglibProxy = new CglibProxy();
        UserDao userDao = new UserDao();
        UserDao userDao1 = (UserDao) cglibProxy.creatProxy(userDao);
        userDao1.say();
        userDao1.listen();
    }

}

                    SpringжҸҗдҫӣдәҶдёҖзі»еҲ—й…ҚзҪ®Spring AOPзҡ„XMLе…ғзҙ гҖӮ

                    е…ғзҙ жҸҸиҝ°
                    <aop:config>SpringAOPй…ҚзҪ®зҡ„ж №е…ғзҙ гҖӮ
                    <aop:aspect>й…ҚзҪ®еҲҮйқўгҖӮ
                    <aop:advisor>й…ҚзҪ®йҖҡзҹҘеҷЁгҖӮ
                    <aop:pointcut>й…ҚзҪ®еҲҮе…ҘзӮ№гҖӮ
                    <aop:before>й…ҚзҪ®еүҚзҪ®йҖҡзҹҘпјҢеңЁзӣ®ж Үж–№жі•жү§иЎҢеүҚе®һж–ҪеўһејәгҖӮ

                    <aop:after-returning>й…ҚзҪ®иҝ”еӣһйҖҡзҹҘпјҢеңЁзӣ®ж Үж–№жі•жҲҗеҠҹжү§иЎҢеҗҺи°ғз”ЁйҖҡзҹҘгҖӮ

                    # й…ҚзҪ®еҲҮйқў

                    <aop:aspect>:й…ҚзҪ®еҲҮйқўдҪҝз”Ёзҡ„е…ғзҙ пјҢе°Ҷе·Іе®ҡд№үеҘҪзҡ„Spring BeanиҪ¬жҚўдёәеҲҮйқўBeanгҖӮеӣ жӯӨеңЁдҪҝз”ЁеүҚйңҖиҰҒе…ҲеңЁй…ҚзҪ®ж–Үд»¶дёӯе®ҡд№үдёҖдёӘжҷ®йҖҡзҡ„Spring BeanгҖӮе®ҡд№үе®ҢжҲҗеҗҺеҸҜд»ҘйҖҡиҝҮrefеұһжҖ§еј•з”ЁиҜҘBeanгҖӮ

                    package com.taytay.demo3;
import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.ProceedingJoinPoint;

public class XmlAdvice {
    public void before(JoinPoint joinPoint){
        System.out.println("еүҚзҪ®дҝЎжҒҜ");
        System.out.println("зӣ®ж Үзұ»жҳҜпјҡ" + joinPoint.getTarget());
        System.out.println("иў«еўһејәзҡ„ж–№жі•жҳҜпјҡ" + joinPoint.getSignature().getName());
    }
    public void afterReturning(JoinPoint joinPoint){
        System.out.println("иҝҷжҳҜиҝ”еӣһйҖҡзҹҘпјҲеңЁдёҚдә§з”ҹејӮеёёж—¶и°ғз”Ёпјү");
        System.out.println("иў«з»Үе…ҘеўһејәеӨ„зҗҶзҡ„зӣ®ж Үж–№жі•дёә" + joinPoint.getSignature().getName());
    }
    public Object around(ProceedingJoinPoint joinPoint) throws Throwable{
        System.out.println("иҝҷжҳҜзҺҜз»•йҖҡзҹҘд№ӢеүҚзҡ„йғЁеҲҶ");
        Object object = joinPoint.proceed();
        System.out.println("иҝҷжҳҜзҺҜз»•йҖҡзҹҘд№ӢеҗҺзҡ„йғЁеҲҶ");
        return object;
    }
    public void afterException(){
        System.out.println("иҝҷжҳҜејӮеёёеҗҺзҡ„йҖҡзҹҘпјҒ");
    }
    public void after(){
        System.out.println("иҝҷжҳҜеҗҺзҪ®йҖҡзҹҘпјҒ");
    }
}

package com.taytay.demo4;

import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.*;

@Aspect
public class AnnoAdvice {
    @Pointcut("execution(* com.taytay.demo3.impl.UserDaoImpl.*(..))")
    public void pointcut(){
    }
    @Before("pointcut()")
    public void before(JoinPoint joinPoint){
        System.out.println("еүҚзҪ®дҝЎжҒҜ");
        System.out.println("зӣ®ж Үзұ»жҳҜпјҡ" + joinPoint.getTarget());
        System.out.println("иў«еўһејәзҡ„ж–№жі•жҳҜпјҡ" + joinPoint.getSignature().getName());
    }
    @After("pointcut()")
    public void after(){
        System.out.println("иҝҷжҳҜеҗҺзҪ®йҖҡзҹҘпјҒ");
    }
    @AfterReturning("pointcut()")
    public void afterReturning(JoinPoint joinPoint){
        System.out.println("иҝҷжҳҜиҝ”еӣһйҖҡзҹҘпјҲеңЁдёҚдә§з”ҹејӮеёёж—¶и°ғз”Ёпјү");
        System.out.println("иў«з»Үе…ҘеўһејәеӨ„зҗҶзҡ„зӣ®ж Үж–№жі•дёә" + joinPoint.getSignature().getName());
    }
    @Around("pointcut()")
    public Object around(ProceedingJoinPoint joinPoint) throws Throwable{
        System.out.println("иҝҷжҳҜзҺҜз»•йҖҡзҹҘд№ӢеүҚзҡ„йғЁеҲҶ");
        Object object = joinPoint.proceed();
        System.out.println("иҝҷжҳҜзҺҜз»•йҖҡзҹҘд№ӢеҗҺзҡ„йғЁеҲҶ");
        return object;
    }
    @AfterThrowing("pointcut()")
    public void afterException(){
        System.out.println("иҝҷжҳҜејӮеёёеҗҺзҡ„йҖҡзҹҘпјҒ");
    }
}
